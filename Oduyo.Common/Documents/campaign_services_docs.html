public async Task<decimal> CalculatePriceWithCampaign(int packageId, decimal originalPrice)
{
    var currentCampaigns = await _campaignService.GetCurrentCampaignsAsync();
    
    foreach (var campaign in currentCampaigns)
    {
        var isInCampaign = await _campaignPackageService.IsPackageInCampaignAsync(
            campaign.Id, packageId);
        
        if (isInCampaign)
        {
            if (campaign.DiscountRate.HasValue)
                return originalPrice * (1 - campaign.DiscountRate.Value / 100);
            
            if (campaign.DiscountAmount.HasValue)
                return originalPrice - campaign.DiscountAmount.Value;
        }
    }
    
    return originalPrice;
}

public async Task<List<Campaign>> GetCurrentCampaignsCached()
